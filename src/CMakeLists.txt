macro(generic_includes name)
    target_include_directories(${name}
        PUBLIC
        ${STOCHY_BINARY_DIR}
        ${STOCHY_SOURCE_DIR}
        ${CMAKE_CURRENT_BINARY_DIR}
        ${CMAKE_CURRENT_SOURCE_DIR}
    )
endmacro(generic_includes)
